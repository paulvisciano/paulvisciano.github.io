def place_qr_on_card(base_path, out_path, url, margin_right=28, pad=4):
    im = Image.open(base_path).convert('RGB')
    arr = np.array(im)
    h, w = arr.shape[0], arr.shape[1]
    right = arr[:, int(w*0.55):, :]
    lum_right = (0.2126*right[:,:,0] + 0.7152*right[:,:,1] + 0.0722*right[:,:,2]).mean(axis=1)
    threshold = np.percentile(lum_right[int(h*0.4):], 35)
    # find footer top (dark region from bottom)
    y = h-2
    while y>int(h*0.4) and lum_right[y] >= threshold:
        y -= 1
    # now y is within footer dark region; find its upper boundary
    y2 = y
    while y2>int(h*0.4) and lum_right[y2] < threshold:
        y2 -= 1
    footer_top = y2+1
    footer_bottom = h - 18  # just above bottom border
    footer_top_in = footer_top + 10
    footer_bottom_in = footer_bottom - 10
    footer_h = footer_bottom_in - footer_top_in
    center_y = (footer_top_in + footer_bottom_in)//2
    
    # size QR to fit 80% of footer height
    qr_size = int(footer_h * 0.8)
    if qr_size < 64: 
        qr_size = 64  # minimum for scannability
    if qr_size > 220:
        qr_size = 220
    
    # build QR

    im.paste(qr_img, (x, y))

    return out_path, (x,y,qr_size,qr_size), (footer_top, footer_bottom, center_y, footer_h)
